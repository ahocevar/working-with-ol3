goal: give it some style

ol.style


var style = new ol.style.Style({
  symbolizers: [
    new ol.style.Fill({
      color: '#ff0000',
      opacity: 0.5
    }),
    new ol.style.Stroke({
      color: '#ff00ff',
      width: 1.5
    })
  ]});

ol.style.Rule


var style = new ol.style.Style({
  rules: [
    new ol.style.Rule({
      filter: 'type == "highway"',
      symbolizers: [
        new ol.style.Stroke({
          width: 3
        })
      ]
    })
  ]});
